Closures

Why the fuss?

            
function outer() {
   var initialValue = 1;
   function inner(num) {
      return initialValue + num;
   }

   return inner;
}

Alternate Syntax

            
function outer() {
   var initialValue = 1;
   return function (num) {
      return initialValue + num;
   }
}

Function Creators

            
function customGreeter(name) {
   return function(greeting) {
      return greeting + ', ' + name;
   }
}

Scope Preservers

            
function createTeam(teamName) {
   var team = {
      name: teamName,
      players: []
   }

   function createPlayer(playerName){
      team.players.push(playerName)
      return team
   }

   return createPlayer
}

Module Creators

            
function createUser(name) {
   var name = name
   var score = 0
   
   function getScore() {
      return score
   }
   function updateScore(update) {
      score += score
      return score
   }

   return {
      name: name,
      getScore: getScore,
      updateScore: updateScore
   }
}
